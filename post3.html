HomePageCtrl = function($scope, data) {
	var old = new Date(1970,0,1);
	var loop = []
	var d ;
	for( i=0;i<data.length;i++){
		console.log( old.toDateString() , data[i].date.toDateString() );
		if( old < data[i].date ){
			d = data[i];
			old = data[i].date ;
		}
	}
	loop.push(d);
	console.log( d );
	$scope.tags = loop ;
}
